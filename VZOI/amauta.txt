FUNC TIENEDEUDA(CODALUM)
LOCAL CMD
SELECT
AS012006.C_CLIPRO,
SUM(AS012006.DEBE) AS TDB,
SUM(AS012006.HABER) AS THB,
IF(SUM(AS012006.DEBE) IS NULL,0,SUM(AS012006.DEBE) )-IF(SUM(AS012006.HABER) IS NULL,0,SUM(AS012006.HABER)) AS SALDO
FROM AS012006
WHERE LEFT(AS012006.CUENTA,2)='12' AND AS012006.TD='99' AND
GROUP BY AS012006.c_clipro
HAVING  SALDO<>0
ORDER BY AS012006.N_ORDEN,AS012006.N_DECLAR,AS012006.C_CLIPRO

SELECT

 AS012006.C_CLIPRO,CONCAT(alumnos.ape_pat,' ',alumnos.ape_mat,' ',alumnos.nombre)
,AS012006.CONCEPTO__,AS012006.CUO_INI,AS012006.N_DOC,AS012006.FCHNUMER,AS012006.F_VCMTO
,AS012006.CTACTE
,SUM(AS012006.DEBE) AS TDB
,SUM(AS012006.HABER) AS THB
,IF(SUM(AS012006.DEBE) IS NULL,0,SUM(AS012006.DEBE) )-IF(SUM(AS012006.HABER) IS NULL,0,SUM(AS012006.HABER)) AS SALDO
,AS012006.N_DECLAR,AS012006.PAIS_ORIGE,AS012006.CODI_REGI,AS012006.N_ORDEN

FROM AS012006

LEFT JOIN alumnos ON AS012006.C_CLIPRO=alumnos.CODIGO
LEFT JOIN DO012006 ON AS012006.TD=DO012006.TD
WHERE  LEFT(AS012006.CUENTA,2)="12" AND AS012006.TD="99"
GROUP BY AS012006.c_clipro,AS012006.TD,AS012006.N_DOC

HAVING  SALDO<>0

ORDER BY AS012006.C_CLIPRO,AS012006.TD,AS012006.N_DOC
